    position: fixed;
    top: 30px;
    left: 10px;
    cursor: pointer;
    font-size: 150%;

    font-weight: 800;
}

    content: "Up";

<body>
   
<script type="text/javascript">
        
var span = document.createElement('span');
span.setAttribute('id',"str");
var height = 200;
var strollOld;

document.onscroll = function(event){
    if(window.pageYOffset>height){
        span.classList.add('up'),
        span.setAttribute('onclick',"funUp()");
        document.body.appendChild(span);
        strollOld = window.pageYOffset;
    }
    if(window.pageYOffset == 0){
        span.classList.remove('up');    
    }
}

function funUp(){
    window.scrollTo(0,0);
    if(span.classList.contains('up')){
        span.classList.add('down');
        span.setAttribute('onclick',"funDown()");
    }
}

function funDown(){
    window.scrollTo(0, strollOld);
    span.classList.remove('down');
}
    
</script>
